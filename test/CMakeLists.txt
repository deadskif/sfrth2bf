set(SBF2C ${CMAKE_BINARY_DIR}/sbf2c/sbf2c)

macro(FRTH2BF_TARGET Name Input Output)
	#execute_process(COMMAND echo "${Input} ${Output}")
	add_custom_command(OUTPUT ${Output}
	 COMMAND ${SFRTH2BF}
	 ARGS -d ${CMAKE_SOURCE_DIR}/sfrth2bf/dict.4th -o${Output} ${Input}
	 DEPENDS ${Input} 
	 COMMENT "[sfrth2bf][${Name}]"
	 WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
endmacro(FRTH2BF_TARGET)

macro(BF2C_TARGET Name Input Output)
	#execute_process(COMMAND echo "${Input} ${Output}")
	add_custom_command(OUTPUT ${Output}
	 COMMAND ${SBF2C}
	 ARGS -o${Output} ${Input}
	 DEPENDS ${Input}
	 COMMENT "[sbf2c][${Name}]"
	 WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
endmacro(BF2C_TARGET)
